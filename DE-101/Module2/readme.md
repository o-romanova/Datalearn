## SQL запросы на основе экселевского дашборда

select 
	to_char(order_date, 'YYYY-MM') as order_date_by_month,
	round(sum(profit), 1) as profit_current_month
from 
	public.orders
group by
	to_char(order_date, 'YYYY-MM')
order by
	1;
